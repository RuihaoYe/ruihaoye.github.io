基本命令

# new
hexo new [layout](post, page, draft) <title>

# generate
hexo [-d](deploy after generation finishes) generate

# public
hexo publish [layout] <filename>

# server
hexo [-p port] [-s](only server static files) [-l](enable logger) server

# deploy
hexo [-g](generate before deployment)deploy

# render
hexo render [-o](output destination) <file1> [file2] ...

# clean: Cleans the cache file (db.json) and generated files (public).
hexo clean

# list
hexo list <type>(page, post, route, tag, category)

# version
hexo version

# 其他参数
# disalbes loading plugins and scripts
hexo --safe
# debug mode
hexo --debug

tag plugins
