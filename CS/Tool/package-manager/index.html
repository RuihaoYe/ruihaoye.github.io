NPM

# 查询当前镜像
npm get registry
# 设置为淘宝镜像
npm config set registry https://registry.npm.taobao.org/

PNPM

# 查询当前镜像
pnpm get registry
# 设置为淘宝镜像
pnpm config set registry https://registry.npm.taobao.org/

# 添加进dependencies
pnpm add dep
# 添加进devDependencies
pnpm add -D dep
# 添加进optionalDependencies
pnpm add -O dep
# 全局安装
pnpm add --global dep

# 移除依赖
pnpm rm dep
# 移除全局依赖
pnpm rm --global dep

Yarn

# 查询当前镜像
yarn config get registry
# 设置为淘宝镜像
yarn config set registry https://registry.npm.taobao.org/
